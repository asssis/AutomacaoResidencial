Create table Usuario
(
id int identity not null,
login varchar(50),
senha varchar(50),
nome varchar(50),
email varchar(50),
rg varchar(50),
telefone varchar(50),
cpf_cnpj varchar(50),
estado int,
nivel int
)
ALTER TABLE Usuario
		ADD CONSTRAINT PK_USUARIO
		PRIMARY KEY(id)

 Create table Unidade(
 id int identity not null,
 id_usuario int,
 descricao varchar(50),
 bairro varchar(50),
 cep varchar(50),
 cidade varchar(50),
 endereco varchar(50),
 estado varchar(50),
 pais varchar(50),
 tempo varchar(50),
 )
 ALTER TABLE Unidade
		ADD CONSTRAINT PK_UNIDADE
		PRIMARY KEY(id)
		
ALTER TABLE Unidade
        ADD CONSTRAINT FK_UNIDADE_USUARIO	
		FOREIGN KEY (id_usuario)
		REFERENCES Usuario(id)

Create table Divisao
(

ALTER TABLE Divisao

ALTER TABLE Divisao
